create table Sapi (
    id serial primary key,
    nomorRegistrasiTernak varchar(50),
    tanggalTulis date,
    namaPemilik varchar(255),
    tanggalLahir date,
    jenisSapi varchar(50),
    jenisKelamin varchar(50),
    image bytea
);

create table Akun (
    id serial primary key not null,
    email varchar(255) not null unique,
    name varchar(255),
    password varchar(255)
);

create table TanggalBirahi (
    id int,
    tanggal date,
    foreign key(id) references Sapi(id)
);

create table TanggalIB (
    id int,
    tanggal date,
    foreign key(id) references Sapi(id)
);

create table KodeStrawSemen (
    id int,
    kode varchar(50),
    foreign key(id) references Sapi(id)
);

create table TanggalBunting (
    id int,
    tanggal date,
    foreign key(id) references Sapi(id)
);

create table TanggalPemerikasaanKebuntingan (
    id int,
    tanggal date,
    foreign key(id) references Sapi(id)
);

create table Penyakit (
    id int,
    jenis varchar(50),
    tanggal date,
    foreign key(id) references Sapi(id)
);

create table Treatment (
    id int,
    jenis varchar(50),
    tanggal date,
    foreign key(id) references Sapi(id)
);
